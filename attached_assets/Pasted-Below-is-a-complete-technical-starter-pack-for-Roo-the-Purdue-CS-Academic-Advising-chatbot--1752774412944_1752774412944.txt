```python
# scrape_html.py
import requests, pandas as pd
from bs4 import BeautifulSoup
URLS = [
    "https://www.cs.purdue.edu/undergraduate/curriculum/bachelor.html",  # Degree reqs
    "https://catalog.purdue.edu/preview_program.php?catoid=7&poid=6557", # Catalog
    # add semester pages here ‚Ä¶
]
rows = []
for url in URLS:
    soup = BeautifulSoup(requests.get(url, timeout=10).text, "html.parser")
    for tr in soup.select("tr"):
        cols = [td.get_text(" ", strip=True) for td in tr.find_all("td")]
        if len(cols) >= 2 and cols[0].startswith("CS"):
            rows.append(dict(code=cols[0], title=cols[1], source=url))
pd.DataFrame(rows).to_csv("courses_html.csv", index=False)
```

````
## üîå OpenAI & retrieval glue code (minimal)  

```python
import os, json, openai, faiss
from llama_index import VectorStoreIndex, SimpleDirectoryReader

openai.api_key = os.getenv("OPENAI_API_KEY")

# Load index  
index = VectorStoreIndex.load("faiss_index")

def generate(user_msg, chat_history):
    context_chunks = index.similarity_search(user_msg, k=5)
    messages = [
        {"role":"system", "content": open("prompts/system.txt").read()},
        *chat_history,
        {"role":"system", "content": "\n\n".join([c.page_content for c in context_chunks])},
        {"role":"user", "content": user_msg}
    ]
    resp = openai.ChatCompletion.create(
        model="gpt-4o-mini",
        messages=messages,
        temperature=0.4,
        top_p=0.9
    )
    return resp.choices[0].message.content
````

*(This single call keeps your CLI clean.  Save a `history` list just like in the earlier REPL example.)*
